SELECT `e`.`Id` AS `EmployeeId`, `e`.`Name` AS `EmployeeName`, `l`.`Id` AS `LeaveTypeId`, `l`.`Name` AS `LeaveTypeName`, CASE
    WHEN `l0`.`Id` IS NULL THEN 0.0
    ELSE `l0`.`FixedDaysOffBalance`
END AS `FixedDaysOffBalance`, COALESCE((
    SELECT COALESCE(SUM(`l1`.`DurationInDays`), 0.0)
    FROM `LeaveRequests` AS `l1`
    WHERE (`l1`.`EmployeeId` = `e`.`Id`) AND (`l1`.`LeaveTypeId` = `l`.`Id`)), 0.0) AS `DaysTaken`
FROM `Employees` AS `e`
CROSS JOIN `LeaveTypes` AS `l`
LEFT JOIN `LeaveBalances` AS `l0` ON (`e`.`Id` = `l0`.`EmployeeId`) AND (`l`.`Id` = `l0`.`LeaveTypeId`)eId`)
